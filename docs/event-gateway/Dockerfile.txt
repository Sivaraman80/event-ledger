# Stage 1: Build the artifact
FROM maven:3.9-eclipse-temurin-17 AS build
WORKDIR /app
COPY pom.xml .
COPY src ./src
RUN mvn clean package -DskipTests

# Stage 2: Minimalist Runtime deployment container base image
FROM eclipse-temurin:17-jre-jammy
WORKDIR /app
COPY --from=build /app/target/event-gateway-*.jar gateway-service.jar
EXPOSE 8081
ENTRYPOINT ["java", "-jar", "gateway-service.jar"]
